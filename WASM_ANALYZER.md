# WASM Analyzer

## Overview

The WASM Analyzer is a comprehensive analysis and visualization tool for
WebAssembly modules generated by Blitz3D-WASM. It provides deep insights into
compiled code quality, performance characteristics, and potential issues through
interactive visualizations and detailed reports.

### Features

- **Stack Balance Validation**: 3-stack algorithm per WebAssembly specification
- **Type Consistency Analysis**: Validates type safety across operations
- **Control Flow Analysis**: Identifies optimization opportunities
- **Interactive Dashboard**: Real-time visual analysis interface
- **Compare Functionality**: Track changes between builds
- **Performance Metrics**: Detailed instruction and function profiling

## Stack Balance Validation

### 3-Stack Algorithm

The validator implements the WebAssembly specification's 3-stack validation
system:

```typescript
class StackValidator {
  private valueStack: number[] = [];
  private controlStack: ControlFrame[] = [];
  private labelStack: Label[] = [];

  validateInstruction(inst: Instruction): ValidationResult {
    switch (inst.opcode) {
      case OpCode.const_i32:
        this.valueStack.push(Type.i32);
        return { valid: true };

      case OpCode.add:
        return this.validateBinaryOperation(inst, Type.i32);

      case OpCode.if:
        return this.validateIfInstruction(inst);

      case OpCode.loop:
        return this.validateLoopInstruction(inst);

        // ... handle all opcodes
    }
  }

  private validateBinaryOperation(
    inst: Instruction,
    expectedType: Type,
  ): ValidationResult {
    if (this.valueStack.length < 2) {
      return { valid: false, error: "Stack underflow in binary operation" };
    }

    const type1 = this.valueStack.pop()!;
    const type2 = this.valueStack.pop()!;

    if (type1 !== expectedType || type2 !== expectedType) {
      return {
        valid: false,
        error:
          `Type mismatch: expected ${expectedType}, got ${type1}, ${type2}`,
      };
    }

    this.valueStack.push(expectedType);
    return { valid: true };
  }

  private validateIfInstruction(inst: Instruction): ValidationResult {
    // Validate condition type (must be i32)
    if (this.valueStack.length === 0) {
      return { valid: false, error: "Stack underflow in if instruction" };
    }

    const conditionType = this.valueStack.pop()!;
    if (conditionType !== Type.i32) {
      return {
        valid: false,
        error: `If condition must be i32, got ${conditionType}`,
      };
    }

    // Push control frames for branches
    this.controlStack.push({
      type: ControlFrameType.If,
      expectedTypes: [],
      resultTypes: [],
    });

    return { valid: true };
  }
}
```

## Type Analysis

### Type System Implementation

```typescript
enum Type {
  i32 = "i32",
  i64 = "i64",
  f32 = "f32",
  f64 = "f64",
  v128 = "v128",
  func = "func",
  externref = "externref",
}

interface TypeConstraint {
  expected: Type[];
  actual: Type;
  instruction: string;
  line: number;
}

class TypeAnalyzer {
  private constraints: TypeConstraint[] = [];
  private typeMap = new Map<number, Type>();

  analyzeFunction(func: Function): TypeAnalysisReport {
    this.constraints.length = 0;
    this.typeMap.clear();

    // Analyze each instruction in function
    for (let i = 0; i < func.body.length; i++) {
      const inst = func.body[i];
      this.analyzeInstruction(inst, i);
    }

    return {
      constraints: this.constraints,
      violations: this.constraints.filter((c) => !this.isValidConstraint(c)),
      coverage: this.calculateTypeCoverage(func),
      optimizations: this.suggestOptimizations(),
    };
  }

  private analyzeInstruction(inst: Instruction, index: number): void {
    switch (inst.opcode) {
      case OpCode.const_i32:
        this.typeMap.set(index, Type.i32);
        break;

      case OpCode.const_f32:
        this.typeMap.set(index, Type.f32);
        break;

      case OpCode.call:
        this.analyzeCallInstruction(inst, index);
        break;

      case OpCode.select:
        this.analyzeSelectInstruction(inst, index);
        break;
    }
  }

  private analyzeCallInstruction(inst: Instruction, index: number): void {
    const targetFunc = this.resolveFunction(inst.index);
    const paramTypes = targetFunc.parameters;
    const resultTypes = targetFunc.results;

    // Check parameter types
    for (let i = 0; i < paramTypes.length; i++) {
      const actualType = this.popStackType();
      if (actualType !== paramTypes[i]) {
        this.constraints.push({
          expected: [paramTypes[i]],
          actual: actualType,
          instruction: `call ${targetFunc.name}`,
          line: index,
        });
      }
    }

    // Push result types
    for (const resultType of resultTypes) {
      this.pushStackType(resultType);
      this.typeMap.set(index, resultType);
    }
  }
}
```

## Performance Profiling

### Instruction Profiling

```typescript
interface InstructionProfile {
  opcode: string;
  count: number;
  cycles: number;
  frequency: number;
  hotspots: number[];
}

class PerformanceProfiler {
  private profiles = new Map<string, InstructionProfile>();
  private callGraph = new CallGraph();

  analyzeModule(module: WASMModule): PerformanceReport {
    for (const func of module.functions) {
      this.analyzeFunction(func);
    }

    return {
      instructionProfiles: Array.from(this.profiles.values()),
      callGraph: this.callGraph,
      bottlenecks: this.identifyBottlenecks(),
      optimizations: this.suggestOptimizations(),
    };
  }

  private analyzeFunction(func: Function): void {
    const cycles = this.estimateExecutionCycles(func);

    for (const inst of func.body) {
      const profile = this.profiles.get(inst.opcode) || {
        opcode: inst.opcode,
        count: 0,
        cycles: 0,
        frequency: 0,
        hotspots: [],
      };

      profile.count++;
      profile.cycles += this.getInstructionCycles(inst.opcode);
      profile.frequency = profile.count / func.body.length;
      profile.hotspots.push(func.index);

      this.profiles.set(inst.opcode, profile);
    }
  }

  private estimateExecutionCycles(func: Function): number {
    let cycles = 0;

    for (const inst of func.body) {
      cycles += this.getInstructionCycles(inst.opcode);
    }

    return cycles;
  }

  private getInstructionCycles(opcode: string): number {
    const cycleMap: Record<string, number> = {
      "i32.add": 1,
      "i32.mul": 3,
      "i32.div": 25,
      "f32.add": 3,
      "f32.mul": 5,
      "f32.div": 12,
      "local.get": 1,
      "local.set": 1,
      "global.get": 2,
      "global.set": 2,
      "call": 10,
      "br": 1,
      "br_if": 2,
      "loop": 1,
      "if": 1,
    };

    return cycleMap[opcode] || 5; // Default cost
  }
}
```

## Compare Functionality

### Build Comparison

Compare different versions of compiled modules to track changes:

```typescript
interface ComparisonResult {
  baseline: AnalysisData;
  current: AnalysisData;
  changes: {
    instructions: InstructionChange[];
    functions: FunctionChange[];
    performance: PerformanceChange[];
    validation: ValidationChange[];
  };
  summary: {
    totalChanges: number;
    improvements: number;
    regressions: number;
  };
}

class BuildComparator {
  compareBuilds(
    baseline: AnalysisData,
    current: AnalysisData,
  ): ComparisonResult {
    return {
      baseline,
      current,
      changes: {
        instructions: this.compareInstructions(baseline, current),
        functions: this.compareFunctions(baseline, current),
        performance: this.comparePerformance(baseline, current),
        validation: this.compareValidation(baseline, current),
      },
      summary: this.generateSummary(baseline, current),
    };
  }

  private compareInstructions(
    baseline: AnalysisData,
    current: AnalysisData,
  ): InstructionChange[] {
    const changes: InstructionChange[] = [];

    const baselineInstructions = baseline.instructions.distribution;
    const currentInstructions = current.instructions.distribution;

    // Find new instructions
    for (const [opcode, count] of Object.entries(currentInstructions)) {
      if (!baselineInstructions[opcode]) {
        changes.push({
          type: "added",
          opcode,
          count,
          impact: this.estimatePerformanceImpact(opcode, count),
        });
      } else if (baselineInstructions[opcode] !== count) {
        changes.push({
          type: "changed",
          opcode,
          from: baselineInstructions[opcode],
          to: count,
          impact: this.estimatePerformanceDelta(
            opcode,
            count - baselineInstructions[opcode],
          ),
        });
      }
    }

    // Find removed instructions
    for (const opcode of Object.keys(baselineInstructions)) {
      if (!currentInstructions[opcode]) {
        changes.push({
          type: "removed",
          opcode,
          count: baselineInstructions[opcode],
          impact: this.estimatePerformanceImpact(
            opcode,
            -baselineInstructions[opcode],
          ),
        });
      }
    }

    return changes.sort((a, b) => Math.abs(b.impact) - Math.abs(a.impact));
  }

  private estimatePerformanceImpact(opcode: string, delta: number): number {
    const costs: Record<string, number> = {
      "i32.mul": 3,
      "f32.div": 12,
      "call": 10,
      "br": 1,
      "select": 5,
    };

    return (costs[opcode] || 1) * delta;
  }
}
```

## Best Practices

### Analysis Workflow

1. **Pre-commit Analysis**: Run analysis before code commits
2. **Baseline Comparison**: Compare against known-good builds
3. **Performance Regression**: Monitor for performance degradation
4. **Validation Priority**: Address validation errors first
5. **Optimization Focus**: Target high-impact optimizations

### Integration with Development

```bash
# Pre-commit hook (scripts/pre-commit.sh)

  cd Tools/analyzer
```

### Continuous Monitoring

```typescript
// Automated monitoring setup
class ContinuousAnalyzer {
  private baseline: AnalysisData;
  private alertThresholds = {
    performanceRegression: 0.1, // 10% regression
    newValidationErrors: 1,
    instructionGrowth: 0.05, // 5% growth
  };

  async monitorBuild(wasmFile: string): Promise<MonitorReport> {
    const current = await this.analyze(wasmFile);
    const comparison = this.compareWithBaseline(current);

    const alerts = this.generateAlerts(comparison);

    if (alerts.length > 0) {
      await this.sendAlerts(alerts);
    }

    return {
      comparison,
      alerts,
      needsAttention: alerts.length > 0,
    };
  }

  private generateAlerts(comparison: ComparisonResult): Alert[] {
    const alerts: Alert[] = [];

    // Performance regression alert
    if (
      comparison.changes.performance.some((p) =>
        p.delta > this.alertThresholds.performanceRegression
      )
    ) {
      alerts.push({
        type: "performance_regression",
        severity: "warning",
        message: "Performance regression detected in build",
      });
    }

    // New validation errors
    const newErrors = comparison.changes.validation.filter((v) =>
      v.severity === "error"
    );
    if (newErrors.length >= this.alertThresholds.newValidationErrors) {
      alerts.push({
        type: "validation_errors",
        severity: "error",
        message: `${newErrors.length} new validation errors detected`,
      });
    }

    return alerts;
  }
}
```
